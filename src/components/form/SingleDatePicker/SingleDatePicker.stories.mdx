### Blocked days

<Canvas>
    <Story name="blocked days">
        {() => {
            const [date, setDate] = useState(null);
            const handleBlockedDay = (day) => {
                const diff = day.diff(moment(), "days");
                return diff < 0;
            };
            return (
                <SingleDatePicker value={date} onChange={setDate} isDayBlocked={handleBlockedDay} />
            );
        }}
    </Story>
</Canvas>
