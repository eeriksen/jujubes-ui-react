# Form

The `<Form/>` component is used to wrap one or more `<FormItem/>` components.

<ArgsTable of={Form} />

### Example

<Canvas>
    <Story name="basic">
        {() => {
            const [data, setData] = useState({});
            return (
                <Card>
                    <CardContent>
                        <Form onSubmit={() => console.log("SUBMIT")}>
                            <Row>
                                <Col span={24}>
                                    <FormItem label="Your name">
                                        <Input
                                            value={data.name}
                                            onChange={(val) => setData({ ...data, name: val })}
                                            placeholder="Write here..."
                                        />
                                    </FormItem>
                                </Col>
                                <Col span={24}>
                                    <FormItem label="Your comment">
                                        <Textarea
                                            value={data.comment}
                                            onChange={(val) => setData({ ...data, comment: val })}
                                            placeholder="Write here..."
                                        />
                                    </FormItem>
                                </Col>
                                <Col span={24}>
                                    <FormItem>
                                        <Checkbox
                                            checked={data.checked}
                                            onChange={(val) => setData({ ...data, checked: val })}
                                        >
                                            Send to everyone
                                        </Checkbox>
                                    </FormItem>
                                </Col>
                                <Col span={24}>
                                    <FormButtons>
                                        <Button type="submit" color="primary">
                                            Save changes now
                                        </Button>
                                        <Button onClick={() => setData({})}>Delete something</Button>
                                    </FormButtons>
                                </Col>
                            </Row>
                        </Form>
                    </CardContent>
                </Card>
            );
        }}
    </Story>
</Canvas>
