# Form

The `<Form/>` component is used to wrap one or more `<FormItem/>` components.

<ArgsTable of={Form} />

### Example

<Canvas>
    <Story name="basic">
        {() => {
            const [data, setData] = useState({});
            return (
                <Card>
                    <Form onSubmit={() => console.log("SUBMIT")}>
                        <CardContent>
                            <Row>
                                <Col span={12}>
                                    <FormItem label="First name">
                                        <Input
                                            value={data.firstName}
                                            onChange={(val) => setData({ ...data, firstName: val })}
                                            placeholder="John"
                                        />
                                    </FormItem>
                                </Col>
                                <Col span={12}>
                                    <FormItem label="Last name">
                                        <Input
                                            value={data.lastName}
                                            onChange={(val) => setData({ ...data, lastName: val })}
                                            placeholder="Doe"
                                        />
                                    </FormItem>
                                </Col>
                                <Col span={24}>
                                    <FormItem label="Your comment">
                                        <Textarea
                                            emojis
                                            counter
                                            value={data.comment}
                                            onChange={(val) => setData({ ...data, comment: val })}
                                            placeholder="Write here..."
                                        />
                                    </FormItem>
                                </Col>
                                <Col span={24}>
                                    <FormItem>
                                        <Checkbox
                                            checked={data.checked}
                                            onChange={(val) => setData({ ...data, checked: val })}
                                        >
                                            Send to everyone
                                        </Checkbox>
                                    </FormItem>
                                </Col>
                            </Row>
                        </CardContent>
                        <CardContent border="top">
                            <FormButtons>
                                <Button type="submit" color="primary">
                                    Save changes
                                </Button>
                                <Button onClick={() => setData({})} color="silent">
                                    Delete something
                                </Button>
                            </FormButtons>
                        </CardContent>
                    </Form>
                </Card>
            );
        }}
    </Story>
</Canvas>
